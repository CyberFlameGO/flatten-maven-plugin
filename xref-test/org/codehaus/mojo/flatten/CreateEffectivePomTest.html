53   * Test-Case for {@link FlattenMojo}.

57  public class CreateEffectivePomTest
58      extends Assertions
59  {
60  
61      /**
62       * Tests method to create effective POM.
63       *
64       * @throws Exception if something goes wrong.
65       */
66      @Test
67      public void testCreateEffectivePom()
68          throws Exception
69      {
70  
71          String magicValue = "magic-value";
72          Properties userProperties = new Properties();
73          userProperties.setProperty( "cmd.test.property", magicValue );
74  
75          File pomFile = new File( "src/test/resources/cmdpropertysubstituion/pom.xml" );
76          ArtifactRepository localRepository = new MavenArtifactRepository();
77          localRepository.setLayout( new DefaultRepositoryLayout() );
78          ArtifactFactory artifactFactory = new DefaultArtifactFactory();
79          ArtifactHandlerManager artifactHandlerManager = new DefaultArtifactHandlerManager();
80          setDeclaredField( artifactFactory, "artifactHandlerManager", artifactHandlerManager );
81          Map<String, ArtifactHandler> artifactHandlers = new HashMap<String, ArtifactHandler>();
82          setDeclaredField( artifactHandlerManager, "artifactHandlers", artifactHandlers );
83          DefaultDependencyResolver depencencyResolver = new DefaultDependencyResolver();
84          DefaultProjectBuildingRequest projectBuildingRequest = new DefaultProjectBuildingRequest();
85          FlattenModelResolver resolver = new FlattenModelResolver( localRepository, artifactFactory,
86                  depencencyResolver, projectBuildingRequest, Collections.<MavenProject>emptyList() );
87          ModelBuildingRequest buildingRequest =
88              new DefaultModelBuildingRequest().setPomFile( pomFile ).setModelResolver( resolver ).setUserProperties( userProperties );
89          Model effectivePom = FlattenMojo.createEffectivePom( buildingRequest, false );
90          assertThat( effectivePom.getName() ).isEqualTo( magicValue );
91      }
92  
93      private void setDeclaredField( Object pojo, String fieldName, Object propertyValue )
94          throws NoSuchFieldException, IllegalAccessException
95      {
96          Field field = pojo.getClass().getDeclaredField( fieldName );
97          field.setAccessible( true );
98          field.set( pojo, propertyValue );
99      }
