28  /**
29   * Helper class to keep the comments how they have been in the original pom.xml While reading with
30   * {@link org.apache.maven.model.io.xpp3.MavenXpp3Writer} the comments are not placed into the
31   * {@link org.apache.maven.model.Model} and so {@link org.apache.maven.model.io.xpp3.MavenXpp3Writer} is not able to
32   * re-write those comments.
33   * <p>
34   * Workaround (maybe until core is fixed) is to remember all the comments and restore them after MavenXpp3Writer has
35   * created the new flattened pom.xml.
36   * <p>
37   * Current restriction on non-unique child nodes is that this class finds the node back due to the position in the file,
38   * that may lead to mis-re-added comments e.g. on multiple added dependencies (but for e.g. resolveCiFriendliesOnly the
39   * nodes keep stable)
40   */
41  class KeepCommentsInPom
42  {
43  
44      /**
45       * Create an instance with collected current comments of the passed pom.xml file.
46       */
47      static KeepCommentsInPom create( Log aLog, File aOriginalPomFile ) throws MojoExecutionException
48      {
49          KeepCommentsInPom tempKeepCommentsInPom = new KeepCommentsInPom();
50          tempKeepCommentsInPom.setLog( aLog );
51          tempKeepCommentsInPom.loadComments( aOriginalPomFile );
52          return tempKeepCommentsInPom;
53      }

55      private Log log;

57      /**
58       * The unique path list for an original node (the comments are stored via the referenced previousSibling)
59       */
60      private Map<String, Node> commentsPaths;

70      /**
71       * load all current comments and text fragments from xml file
72       *
73       * @param anOriginalPomFile the pom.xml
74       */
75      private void loadComments( File anOriginalPomFile ) throws MojoExecutionException
76      {
77          commentsPaths = new HashMap<>();
78          DocumentBuilderFactory tempDBF = DocumentBuilderFactory.newInstance();
79          DocumentBuilder tempDB;
80          try
81          {
82              tempDB = tempDBF.newDocumentBuilder();
83              Document tempPom = tempDB.parse( anOriginalPomFile );
84              Node tempNode = tempPom.getDocumentElement();
85              walkOverNodes( tempNode, ".", ( node, nodePath ) ->
86              {
87                  // collectNodesByPathNames
88                  commentsPaths.put( nodePath, node );
89              } );
90          }
91          catch ( ParserConfigurationException | SAXException | IOException e )
92          {
93              throw new MojoExecutionException( "Cannot load comments from " + anOriginalPomFile, e );
94          }
95      }
96  
97      /**
98       * Walk over the pom hierarchy of the Document.
99       *
100      * @param Node      the current Node
101      * @param String    the unique path in the parent
102      * @param aConsumer Function to be called with the toBeCollected/found node.
103      */
104     private void walkOverNodes( Node aNode, String aParentPath, BiConsumer<Node, String> aConsumer )
105     {
106         String tempNodeName = aNode.getNodeName();
107         if ( log.isDebugEnabled() )
108         {
109             log.debug( "walkOverNodes: aParentPath=" + aParentPath + " tempNodeName=" + tempNodeName );
110         }
111         String tempNodePath = aParentPath + "\t" + tempNodeName;
112         aConsumer.accept( aNode, tempNodePath );
113         NodeList tempChilds = aNode.getChildNodes();
114         // Copy the childs as aConsumer may change the node sequence (add a comment)
115         List<Node> tempCopiedChilds = new ArrayList<>();
116         Map<String, Integer> tempChildWithSameName = new HashMap<>();
117         for ( int i = 0; i < tempChilds.getLength(); i++ )
118         {
119             Node tempItem = tempChilds.item( i );
120             if ( tempItem.getNodeType() != Node.TEXT_NODE && tempItem.getNodeType() != Node.COMMENT_NODE )
121             {
122                 // Take real nodes to find them back by number
123                 String tempChildNodeName = tempItem.getNodeName();
124                 Integer tempChildWithSameNameCount = tempChildWithSameName.get( tempChildNodeName );
125                 if ( tempChildWithSameNameCount == null )
126                 {
127                     tempChildWithSameNameCount = 1;
128                 }
129                 else
130                 {
131                     tempChildWithSameNameCount += 1;
132                 }
133                 tempChildWithSameName.put( tempChildNodeName, tempChildWithSameNameCount );
134                 tempCopiedChilds.add( tempItem );
135             }
136         }
137         Map<String, Integer> tempChildWithSameNameCounters = new HashMap<>();
138         for ( int i = 0; i < tempCopiedChilds.size(); i++ )
139         {
140             Node tempCopiedChild = tempCopiedChilds.get( i );
141             String tempChildNodeName = tempCopiedChild.getNodeName();
142             if ( tempChildWithSameName.get( tempChildNodeName ) > 1 )
143             {
144                 Integer tempChildWithSameNameCounter = tempChildWithSameNameCounters.get( tempChildNodeName );
145                 if ( tempChildWithSameNameCounter == null )
146                 {
147                     tempChildWithSameNameCounter = 1;
148                 }
149                 else
150                 {
151                     tempChildWithSameNameCounter += 1;
152                 }
153                 tempChildWithSameNameCounters.put( tempChildNodeName, tempChildWithSameNameCounter );
154                 // add a counter to find back the correct node.
155                 walkOverNodes( tempCopiedChild, tempNodePath + "\t" + tempChildWithSameNameCounter, aConsumer );
156             }
157             else
158             {
159                 // unique child names
160                 walkOverNodes( tempCopiedChild, tempNodePath, aConsumer );
161             }
162         }
163     }

165     /**
166      * @param String the XML written by {@link org.apache.maven.model.io.xpp3.MavenXpp3Writer}
167      */
168     public String restoreOriginalComments( String anXml, String aModelEncoding ) throws MojoExecutionException
169     {
170         DocumentBuilderFactory tempDBF = DocumentBuilderFactory.newInstance();
171         DocumentBuilder tempDB;
172         try
173         {
174             tempDB = tempDBF.newDocumentBuilder();
175             String tempEncoding = aModelEncoding == null ? "UTF-8" : aModelEncoding; // default encoding UTF-8 when
176             // nothing in pom model.
177             Document tempPom = tempDB.parse( new ByteArrayInputStream( anXml.getBytes( tempEncoding ) ) );
178             Node tempNode = tempPom.getDocumentElement();
179             walkOverNodes( tempNode, ".", ( newNode, nodePath ) ->
180             {
181                 Node tempOriginalNode = commentsPaths.get( nodePath );
182                 if ( tempOriginalNode != null )
183                 {
184                     String tempOriginalNodeName = tempOriginalNode.getNodeName();
185                     if ( tempOriginalNodeName.equals( newNode.getNodeName() ) )
186                     {
187                         // found matching node
188                         Node tempRefChild = newNode;
189                         Node tempPotentialCommentOrText = tempOriginalNode.getPreviousSibling();
190                         while ( tempPotentialCommentOrText != null
191                             && tempPotentialCommentOrText.getNodeType() == Node.TEXT_NODE )
192                         {
193                             // skip text in the original xml node
194                             tempPotentialCommentOrText = tempPotentialCommentOrText.getPreviousSibling();
195                         }
196                         while ( tempPotentialCommentOrText != null
197                             && tempPotentialCommentOrText.getNodeType() == Node.COMMENT_NODE )
198                         {
199                             // copy the node to be able to call previoussibling for next element
200                             Node tempRefPrevious = tempRefChild.getPreviousSibling();
201                             String tempWhitespaceTextBeforeRefNode = null;
202                             if ( tempRefPrevious != null && tempRefPrevious.getNodeType() == Node.TEXT_NODE )
203                             {
204                                 tempWhitespaceTextBeforeRefNode = tempRefPrevious.getNodeValue();
205                             }
206                             Node tempNewComment;
207                             tempNewComment = tempPom.createComment( tempPotentialCommentOrText.getNodeValue() );
208                             tempRefChild.getParentNode().insertBefore( tempNewComment, tempRefChild );
209                             // copy the whitespaces between comment and refNode
210                             if ( tempWhitespaceTextBeforeRefNode != null )
211                             {
212                                 tempRefChild.getParentNode().insertBefore(
213                                     tempPom.createTextNode( tempWhitespaceTextBeforeRefNode ), tempRefChild );
214                             }
215 
216                             tempRefChild = tempNewComment;
217 
218                             tempPotentialCommentOrText = tempPotentialCommentOrText.getPreviousSibling();
219                             while ( tempPotentialCommentOrText != null
220                                 && tempPotentialCommentOrText.getNodeType() == Node.TEXT_NODE )
221                             {
222                                 // skip text in the original xml node
223                                 tempPotentialCommentOrText = tempPotentialCommentOrText.getPreviousSibling();
224                             }
225                         }
226                     }
227                 }
228             } );
229             return writeDocumentToString( tempPom );
230         }
231         catch ( ParserConfigurationException | SAXException | IOException | ClassNotFoundException
232                 | InstantiationException | IllegalAccessException | ClassCastException e )
233         {
234             throw new MojoExecutionException( "Cannot add  comments", e );
235         }
236     }

238     /**
239      * Use an LSSerializer to keep whitespaces added by MavenXpp3Writer
240      *
241      * @param Document the pom to write to String.
242      */
243     private String writeDocumentToString( Document aPom )
244         throws ClassNotFoundException, InstantiationException, IllegalAccessException
245     {
246         DOMImplementationRegistry registry = DOMImplementationRegistry.newInstance();
247         DOMImplementationLS impl = (DOMImplementationLS) registry.getDOMImplementation( "LS" );
248         LSOutput output = impl.createLSOutput();
249         output.setEncoding( "UTF-8" );
250         ByteArrayOutputStream outStream = new ByteArrayOutputStream();
251         output.setByteStream( outStream );
252         LSSerializer writer = impl.createLSSerializer();
253         writer.write( aPom, output );
254         return new String( outStream.toByteArray() );
255     }
