48  /**
49   * This class reflects a property of a {@link Model POM}. It contains {@link #getPomProperties() all available
50   * properties} as constants and allows generic access to {@link #get(Model) read} and {@link #set(Model, Object) write}
51   * the property via a {@link Model}.
52   *
53   * @param <V> is the generic type of the {@link #get(Model) property value}.
54   * @author Joerg Hohwiller (hohwille at users.sourceforge.net)
55   * @since 1.0.0-beta-4
56   */
57  public abstract class PomProperty<V>
58  {

414     /** @see Build#getPluginManagement() */
415     public static final PomProperty<PluginManagement> PLUGIN_MANAGEMENT =
416         new PomProperty<PluginManagement>( "pluginManagement", PluginManagement.class )
417         {
418             @Override
419             public PluginManagement get( Model model )
420             {
421                 if ( model.getBuild() == null )
422                 {
423                     return null;
424                 }
425                 return model.getBuild().getPluginManagement();
426             }
427 
428             @Override
429             public void set( Model model, PluginManagement value )
430             {
431                 if ( model.getBuild() == null )
432                 {
433                     model.setBuild( new Build() );
434                 }
435                 model.getBuild().setPluginManagement( value );
436             };
437         };

607     private static final PomProperty<?>[] POM_PROPERTIES_ARRAY = new PomProperty<?>[] { ARTIFACT_ID, BUILD,
608         CI_MANAGEMENT, CONTRIBUTORS, DEPENDENCIES, DEPENDENCY_MANAGEMENT, DESCRIPTION, DEVELOPERS,
609         DISTRIBUTION_MANAGEMENT, GROUP_ID, INCEPTION_YEAR, ISSUE_MANAGEMENT, LICENSES, MAILING_LISTS, MODEL_ENCODING,
610         MODEL_VERSION, MODULES, NAME, ORGANIZATION, PACKAGING, PARENT, PLUGIN_MANAGEMENT, PLUGIN_REPOSITORIES, POM_FILE,
611         PREREQUISITES, PROFILES, PROPERTIES, REPORTING, REPOSITORIES, SCM, URL, VERSION };
612 
613     private static final List<PomProperty<?>> POM_PROPERTIES =
614         Collections.unmodifiableList( Arrays.asList( POM_PROPERTIES_ARRAY ) );
615 
616     private final String name;
617 
618     private final Class<V> valueType;
619 
620     /**
621      * The constructor.
622      *
623      * @param name - see {@link #getName()}.
624      * @param valueType - see {@link #getValueType()}.
625      */
626     public PomProperty( String name, Class<V> valueType )
627     {
628         super();
629         this.name = name;
630         this.valueType = valueType;
631     }
632 
633     /**
634      * @return name
635      */
636     public String getName()
637     {
638         return this.name;
639     }
640 
641     /**
642      * @return {@link Class} reflecting the type of the {@link #get(Model) property value}.
643      */
644     public Class<V> getValueType()
645     {
646         return this.valueType;
647     }
648 
649     /**
650      * @return <code>true</code> if required for flattened POM, <code>false</code> otherwise.
651      */
652     public boolean isRequired()
653     {
654         if ( this == GROUP_ID || this == ARTIFACT_ID || this == VERSION )
655         {
656             return true;
657         }
658         return false;
659     }
660 
661     /**
662      * @return <code>true</code> if this property represents an XML element of the POM representation,
663      *         <code>false</code> otherwise (if an internal property such as {@link Model#getPomFile()}).
664      */
665     public boolean isElement()
666     {
667 
668         if ( this == POM_FILE )
669         {
670             return false;
671         }
672         if ( this == MODEL_ENCODING )
673         {
674             return false;
675         }
676         return true;
677     }
678 
679     /**
680      * Generic getter for reading a {@link PomProperty} from a {@link Model}.
681      *
682      * @param model is the {@link Model} to read from.
683      * @return the value of the property to read identified by this {@link PomProperty}.
684      */
685     public abstract V get( Model model );
686 
687     /**
688      * Generic setter for writing a {@link PomProperty} in a {@link Model}.
689      *
690      * @param model is the {@link Model} to write to.
691      * @param value is the value of the property to write identified by this {@link PomProperty}.
692      */
693     public abstract void set( Model model, V value );
694 
695     /**
696      * Copies the value identified by this {@link PomProperty} from the given <code>source</code> {@link Model} to the
697      * given <code>target</code> {@link Model}.
698      *
699      * @param source is the {@link Model} to copy from (read).
700      * @param target is the {@link Model} to copy to (write).
701      */
702     public void copy( Model source, Model target )
703     {
704 
705         V value = get( source );
706         if ( value != null )
707         {
708             set( target, value );
709         }
710     }
711 
712     /**
713      * @return an {@link Collections#unmodifiableList(List) unmodifiable} {@link List} with all {@link PomProperty
714      *         properties}.
715      */
716     public static List<PomProperty<?>> getPomProperties()
717     {
718         return POM_PROPERTIES;
719     }
