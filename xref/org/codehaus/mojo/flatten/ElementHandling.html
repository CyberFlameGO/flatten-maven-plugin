3   /**
4    * This enum contains each available handling for a POM element when flattening.
5    *
6    * @author Joerg Hohwiller (hohwille at users.sourceforge.net)
7    * @since 1.0.0-beta-4
8    */
9   public enum ElementHandling
10  {
11  
12      /**
13       * Flatten the element. For most elements this means that they will be removed and are not present in the flattened
14       * POM.
15       */
16      flatten,
17  
18      /** Take the element from the effective POM. */
19      expand,
20  
21      /** Take the element from the resolved POM. */
22      resolve,
23  
24      /** Take the element from the interpolated POM (original POM with variables interpolated). */
25      interpolate,
26  
27      /** Take the element untouched from the original POM. */
28      keep,
29  
30      /** Remove the element entirely so it will not be present in flattened POM. */
31      remove
32  
33  }
