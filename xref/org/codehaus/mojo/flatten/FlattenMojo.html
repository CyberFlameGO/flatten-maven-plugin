94  /**
95   * This MOJO realizes the goal <code>flatten</code> that generates the flattened POM and {@link #isUpdatePomFile()
96   * potentially updates the POM file} so that the current {@link MavenProject}'s {@link MavenProject#getFile() file}
97   * points to the flattened POM instead of the original <code>pom.xml</code> file. The flattened POM is a reduced version
98   * of the original POM with the focus to contain only the important information for consuming it. Therefore information
99   * that is only required for maintenance by developers and to build the project artifact(s) are stripped. Starting from
100  * here we specify how the flattened POM is created from the original POM and its project:<br>
101  * <table border="1" summary="">
102  * <tr>
103  * <th>Element</th>
104  * <th>Transformation</th>
105  * <th>Note</th>
106  * </tr>
107  * <tr>
108  * <td>{@link Model#getModelVersion() modelVersion}</td>
109  * <td>Fixed to "4.0.0"</td>
110  * <td>New maven versions will once be able to evolve the model version without incompatibility to older versions if
111  * flattened POMs get deployed.</td>
112  * </tr>
113  * <tr>
114  * <td>{@link Model#getGroupId() groupId}<br>
115  * {@link Model#getArtifactId() artifactId}<br>
116  * {@link Model#getVersion() version}<br>
117  * {@link Model#getPackaging() packaging}</td>
118  * <td>resolved</td>
119  * <td>copied to the flattened POM but with inheritance from {@link Model#getParent() parent} as well as with all
120  * variables and defaults resolved. These elements are technically required for consumption.</td>
121  * </tr>
122  * <tr>
123  * <td>{@link Model#getLicenses() licenses}</td>
124  * <td>resolved</td>
125  * <td>copied to the flattened POM but with inheritance from {@link Model#getParent() parent} as well as with all
126  * variables and defaults resolved. The licenses would not be required in flattened POM. However, they make sense for
127  * publication and deployment and are important for consumers of your artifact.</td>
128  * </tr>
129  * <tr>
130  * <td>{@link Model#getDependencies() dependencies}</td>
131  * <td>resolved specially</td>
132  * <td>flattened POM contains the actual dependencies of the project. Test dependencies are removed. Variables and
133  * {@link Model#getDependencyManagement() dependencyManagement} is resolved to get fixed dependency attributes
134  * (especially {@link Dependency#getVersion() version}). If {@link #isEmbedBuildProfileDependencies()
135  * embedBuildProfileDependencies} is set to <code>true</code>, then also build-time driven {@link Profile}s will be
136  * evaluated and may add {@link Dependency dependencies}. For further details see {@link Profile}s below.</td>
137  * </tr>
138  * <tr>
139  * <td>{@link Model#getProfiles() profiles}</td>
140  * <td>resolved specially</td>
141  * <td>only the {@link Activation} and the {@link Dependency dependencies} of a {@link Profile} are copied to the
142  * flattened POM. If you set the parameter {@link #isEmbedBuildProfileDependencies() embedBuildProfileDependencies} to
143  * <code>true</code> then only profiles {@link Activation activated} by {@link Activation#getJdk() JDK} or
144  * {@link Activation#getOs() OS} will be added to the flattened POM while the other profiles are triggered by the
145  * current build setup and if activated their impact on dependencies is embedded into the resulting flattened POM.</td>
146  * </tr>
147  * <tr>
148  * <td>{@link Model#getName() name}<br>
149  * {@link Model#getDescription() description}<br>
150  * {@link Model#getUrl() url}<br>
151  * {@link Model#getInceptionYear() inceptionYear}<br>
152  * {@link Model#getOrganization() organization}<br>
153  * {@link Model#getScm() scm}<br>
154  * {@link Model#getDevelopers() developers}<br>
155  * {@link Model#getContributors() contributors}<br>
156  * {@link Model#getMailingLists() mailingLists}<br>
157  * {@link Model#getPluginRepositories() pluginRepositories}<br>
158  * {@link Model#getIssueManagement() issueManagement}<br>
159  * {@link Model#getCiManagement() ciManagement}<br>
160  * {@link Model#getDistributionManagement() distributionManagement}</td>
161  * <td>configurable</td>
162  * <td>Will be stripped from the flattened POM by default. You can configure all of the listed elements inside
163  * <code>pomElements</code> that should be kept in the flattened POM (e.g. {@literal
164  * <pomElements><name/><description/><developers/><contributors/></pomElements>}). For common use-cases there are
165  * predefined modes available via the parameter <code>flattenMode</code> that should be used in preference.</td>
166  * </tr>
167  * <tr>
168  * <td>{@link Model#getPrerequisites() prerequisites}</td>
169  * <td>configurable</td>
170  * <td>Like above but by default NOT removed if packaging is "maven-plugin".</td>
171  * </tr>
172  * <tr>
173  * <td>{@link Model#getRepositories() repositories}</td>
174  * <td>configurable</td>
175  * <td>Like two above but by default NOT removed. If you want have it removed, you need to use the parameter
176  * <code>pomElements</code> and configure the child element <code>repositories</code> with value <code>flatten</code>.
177  * </td>
178  * </tr>
179  * <tr>
180  * <td>{@link Model#getParent() parent}<br>
181  * {@link Model#getBuild() build}<br>
182  * {@link Model#getDependencyManagement() dependencyManagement}<br>
183  * {@link Model#getProperties() properties}<br>
184  * {@link Model#getModules() modules}<br>
185  * {@link Model#getReporting() reporting}</td>
186  * <td>configurable</td>
187  * <td>These elements should typically be completely stripped from the flattened POM. However for ultimate flexibility
188  * (e.g. if you only want to resolve variables in a POM with packaging pom) you can also configure to keep these
189  * elements. We strictly recommend to use this feature with extreme care and only if packaging is pom (for "Bill of
190  * Materials"). In the latter case you configure the parameter <code>flattenMode</code> to the value
191  * <code>bom</code>.<br>
192  * If the <code>build</code> element contains plugins in the <code>build/plugins</code> section which are configured to
193  * load <a href="http://maven.apache.org/pom.html#Extensions">extensions</a>, a reduced <code>build</code> element
194  * containing these plugins will be kept in the flattened pom.</td>
195  * </tr>
196  * </table>
197  *
198  * @author Joerg Hohwiller (hohwille at users.sourceforge.net)
199  */

385     /**
386      * {@inheritDoc}
387      */
388     public void execute()
389         throws MojoExecutionException, MojoFailureException
390     {
391 
392         getLog().info( "Generating flattened POM of project " + this.project.getId() + "..." );
393 
394         File originalPomFile = this.project.getFile();
395         Model flattenedPom = createFlattenedPom( originalPomFile );
396         String headerComment = extractHeaderComment( originalPomFile );
397 
398         File flattenedPomFile = getFlattenedPomFile();
399         writePom( flattenedPom, flattenedPomFile, headerComment );
400 
401         if ( isUpdatePomFile() )
402         {
403             this.project.setPomFile( flattenedPomFile );
404         }
405     }

407     /**
408      * This method extracts the XML header comment if available.
409      *
410      * @param xmlFile is the XML {@link File} to parse.
411      * @return the XML comment between the XML header declaration and the root tag or <code>null</code> if NOT
412      * available.
413      * @throws MojoExecutionException if anything goes wrong.
414      */
415     protected String extractHeaderComment( File xmlFile )
416         throws MojoExecutionException
417     {
418 
419         try
420         {
421             SAXParser parser = SAXParserFactory.newInstance().newSAXParser();
422             SaxHeaderCommentHandler handler = new SaxHeaderCommentHandler();
423             parser.setProperty( "http://xml.org/sax/properties/lexical-handler", handler );
424             parser.parse( xmlFile, handler );
425             return handler.getHeaderComment();
426         }
427         catch ( Exception e )
428         {
429             throw new MojoExecutionException( "Failed to parse XML from " + xmlFile, e );
430         }
431     }

433     /**
434      * Writes the given POM {@link Model} to the given {@link File}.
435      *
436      * @param pom           the {@link Model} of the POM to write.
437      * @param pomFile       the {@link File} where to write the given POM will be written to. {@link File#getParentFile()
438      *                      Parent directories} are {@link File#mkdirs() created} automatically.
439      * @param headerComment is the content of a potential XML comment at the top of the XML (after XML declaration and
440      *                      before root tag). May be <code>null</code> if not present and to be omitted in target POM.
441      * @throws MojoExecutionException if the operation failed (e.g. due to an {@link IOException}).
442      */
443     protected void writePom( Model pom, File pomFile, String headerComment )
444         throws MojoExecutionException
445     {
446 
447         File parentFile = pomFile.getParentFile();
448         if ( !parentFile.exists() )
449         {
450             boolean success = parentFile.mkdirs();
451             if ( !success )
452             {
453                 throw new MojoExecutionException( "Failed to create directory " + pomFile.getParent() );
454             }
455         }
456         // MavenXpp3Writer could internally add the comment but does not expose such feature to API!
457         // Instead we have to write POM XML to String and do post processing on that :(
458         MavenXpp3Writer pomWriter = new MavenXpp3Writer();
459         StringWriter stringWriter = new StringWriter( INITIAL_POM_WRITER_SIZE );
460         try
461         {
462             pomWriter.write( stringWriter, pom );
463         }
464         catch ( IOException e )
465         {
466             throw new MojoExecutionException( "Internal I/O error!", e );
467         }
468         StringBuffer buffer = stringWriter.getBuffer();
469         if ( !StringUtils.isEmpty( headerComment ) )
470         {
471             int projectStartIndex = buffer.indexOf( "<project" );
472             if ( projectStartIndex >= 0 )
473             {
474                 buffer.insert( projectStartIndex, "<!--" + headerComment + "-->\n" );
475             }
476             else
477             {
478                 getLog().warn( "POM XML post-processing failed: no project tag found!" );
479             }
480         }
481         writeStringToFile( buffer.toString(), pomFile, pom.getModelEncoding() );
482     }
483 
484     /**
485      * Writes the given <code>data</code> to the given <code>file</code> using the specified <code>encoding</code>.
486      *
487      * @param data     is the {@link String} to write.
488      * @param file     is the {@link File} to write to.
489      * @param encoding is the encoding to use for writing the file.
490      * @throws MojoExecutionException if anything goes wrong.
491      */
492     protected void writeStringToFile( String data, File file, String encoding )
493         throws MojoExecutionException
494     {
495 
496         byte[] binaryData;
497 
498         try
499         {
500             binaryData = data.getBytes( encoding );
501 
502             if ( file.isFile() && file.canRead() && file.length() == binaryData.length )
503             {
504                 try (InputStream inputStream = new FileInputStream( file ))
505                 {
506                     byte[] buffer = new byte[binaryData.length];
507                     inputStream.read( buffer );
508                     if ( Arrays.equals( buffer, binaryData ) )
509                     {
510                         getLog().debug( "Arrays.equals( buffer, binaryData ) " );
511                         return;
512                     }
513                     getLog().debug( "Not Arrays.equals( buffer, binaryData ) " );
514                 }
515                 catch ( IOException e )
516                 {
517                     // ignore those exceptions, we will overwrite the file
518                     getLog().debug( "Issue reading file: " + file.getPath(), e );
519                 }
520             }
521             else
522             {
523                 getLog().debug( "file: " + file + ",file.length(): " + file.length() + ", binaryData.length: " + binaryData.length );
524             }
525         } catch ( IOException e )
526         {
527             throw new MojoExecutionException( "cannot read String as bytes" , e );
528         }
529         try (OutputStream outStream = new FileOutputStream( file ))
530         {
531             outStream.write( binaryData );
532         }
533         catch ( IOException e )
534         {
535             throw new MojoExecutionException( "Failed to write to " + file, e );
536         }
537     }

539     /**
540      * This method creates the flattened POM what is the main task of this plugin.
541      *
542      * @param pomFile is the name of the original POM file to read and transform.
543      * @return the {@link Model} of the flattened POM.
544      * @throws MojoExecutionException if anything goes wrong (e.g. POM can not be processed).
545      * @throws MojoFailureException   if anything goes wrong (logical error).
546      */
547     protected Model createFlattenedPom( File pomFile ) throws MojoExecutionException, MojoFailureException
548     {
549 
550         ModelBuildingRequest buildingRequest = createModelBuildingRequest( pomFile );
551         Model effectivePom = createEffectivePom( buildingRequest, isEmbedBuildProfileDependencies(), this.flattenMode );
552 
553         Model flattenedPom = new Model();
554 
555         // keep original encoding (we could also normalize to UTF-8 here)
556         String modelEncoding = effectivePom.getModelEncoding();
557         if ( StringUtils.isEmpty( modelEncoding ) )
558         {
559             modelEncoding = "UTF-8";
560         }
561         flattenedPom.setModelEncoding( modelEncoding );
562 
563         Model cleanPom = null;
564         try {
565             cleanPom = createCleanPom( effectivePom );
566         }
567         catch (Exception e) {
568             throw new MojoExecutionException("failed to create a clean pom", e);
569         }
570 
571         FlattenDescriptor descriptor = getFlattenDescriptor();
572         Model originalPom = this.project.getOriginalModel();
573         Model resolvedPom = this.project.getModel();
574         Model interpolatedPom = createResolvedPom( buildingRequest );
575 
576         // copy the configured additional POM elements...
577 
578         for ( PomProperty<?> property : PomProperty.getPomProperties() )
579         {
580             if ( property.isElement() )
581             {
582                 Model sourceModel = getSourceModel( descriptor, property, effectivePom, originalPom, resolvedPom,
583                                                     interpolatedPom, cleanPom );

586                     if ( property.isRequired() )

588                         throw new MojoFailureException( "Property " + property.getName()
589                             + " is required and can not be removed!" );
590                     }
591                 }
592                 else
593                 {
594                     property.copy( sourceModel, flattenedPom );
595                 }
596             }
597         }
598 
599         return flattenedPom;
600     }

614     /**
615      * This method creates the clean POM as a {@link Model} where to copy elements from that shall be
616      * {@link ElementHandling#flatten flattened}. Will be mainly empty but contains some the minimum elements that have
617      * to be kept in flattened POM.
618      *
619      * @param effectivePom is the effective POM.
620      * @return the clean POM.
621      * @throws MojoExecutionException if anything goes wrong.
622      */
623     protected Model createCleanPom( Model effectivePom ) throws MojoExecutionException
624     {
625         Model cleanPom = new Model();
626 
627         cleanPom.setGroupId( effectivePom.getGroupId() );
628         cleanPom.setArtifactId( effectivePom.getArtifactId() );
629         cleanPom.setVersion( effectivePom.getVersion() );
630         cleanPom.setPackaging( effectivePom.getPackaging() );
631         cleanPom.setLicenses( effectivePom.getLicenses() );
632         // fixed to 4.0.0 forever :)
633         cleanPom.setModelVersion( "4.0.0" );
634 
635         // plugins with extensions must stay
636         Build build = effectivePom.getBuild();
637         if ( build != null )
638         {
639             for ( Plugin plugin : build.getPlugins() )
640             {
641                 if ( plugin.isExtensions() )
642                 {
643                     Build cleanBuild = cleanPom.getBuild();
644                     if ( cleanBuild == null )
645                     {
646                         cleanBuild = new Build();
647                         cleanPom.setBuild( cleanBuild );
648                     }
649                     Plugin cleanPlugin = new Plugin();
650                     cleanPlugin.setGroupId( plugin.getGroupId() );
651                     cleanPlugin.setArtifactId( plugin.getArtifactId() );
652                     cleanPlugin.setVersion( plugin.getVersion() );
653                     cleanPlugin.setExtensions( true );
654                     cleanBuild.addPlugin( cleanPlugin );
655                 }
656             }
657         }
658 
659         // transform profiles...
660         for ( Profile profile : effectivePom.getProfiles() )
661         {
662             if ( !isEmbedBuildProfileDependencies() || !isBuildTimeDriven( profile.getActivation() ) )
663             {
664                 if ( !isEmpty( profile.getDependencies() ) || !isEmpty( profile.getRepositories() ) )
665                 {
666                     Profile strippedProfile = new Profile();
667                     strippedProfile.setId( profile.getId() );
668                     strippedProfile.setActivation( profile.getActivation() );
669                     strippedProfile.setDependencies( profile.getDependencies() );
670                     strippedProfile.setRepositories( profile.getRepositories() );
671                     cleanPom.addProfile( strippedProfile );
672                 }
673             }
674         }
675 
676         // transform dependencies...
677         List<Dependency> dependencies = createFlattenedDependencies( effectivePom );
678         cleanPom.setDependencies( dependencies );
679         return cleanPom;
680     }

682     private Model getSourceModel( FlattenDescriptor descriptor, PomProperty<?> property, Model effectivePom,
683                                   Model originalPom, Model resolvedPom, Model interpolatedPom, Model cleanPom )
684     {
685 
686         ElementHandling handling = descriptor.getHandling( property );
687         getLog().debug( "Property " + property.getName() + " will be handled using " + handling
688             + " in flattened POM." );
689         switch ( handling )
690         {
691             case expand:
692                 return effectivePom;
693             case keep:
694                 return originalPom;
695             case resolve:
696                 return resolvedPom;
697             case interpolate:
698                 return interpolatedPom;
699             case flatten:
700                 return cleanPom;
701             case remove:
702                 return null;
703             default:
704                 throw new IllegalStateException( handling.toString() );
705         }
706     }

733     private FlattenDescriptor getFlattenDescriptor()
734         throws MojoFailureException
735     {
736         FlattenDescriptor descriptor = this.pomElements;
737         if ( descriptor == null )
738         {
739             FlattenMode mode = this.flattenMode;
740             if ( mode == null )
741             {
742                 mode = FlattenMode.defaults;
743             }
744             else if ( this.flattenMode == FlattenMode.minimum )
745             {
746                 getLog().warn( "FlattenMode " + FlattenMode.minimum + " is deprecated!" );
747             }
748             descriptor = mode.getDescriptor();
749             if ( "maven-plugin".equals( this.project.getPackaging() ) )
750             {
751                 descriptor.setPrerequisites( ElementHandling.expand );
752             }
753         }
754         else
755         {
756             if ( descriptor.isEmpty() )
757             {
758                 // legacy approach...
759                 // Can't use Model itself as empty elements are never null, so you can't recognize if it was set or not
760                 Xpp3Dom rawDescriptor = this.mojoExecution.getConfiguration().getChild( "pomElements" );
761                 descriptor = new FlattenDescriptor( rawDescriptor );
762             }
763             if ( this.flattenMode != null )
764             {
765                 descriptor = descriptor.merge( this.flattenMode.getDescriptor() );
766             }
767         }
768         return descriptor;
769     }

956     /**
957      * Creates the {@link List} of {@link Dependency dependencies} for the flattened POM. These are all resolved
958      * {@link Dependency dependencies} except for those added from {@link Profile profiles}.
959      *
960      * @param effectiveModel is the effective POM {@link Model} to process.
961      * @return the {@link List} of {@link Dependency dependencies}.
962      * @throws MojoExecutionException if anything goes wrong.
963      */
964     protected List<Dependency> createFlattenedDependencies( Model effectiveModel )
965             throws MojoExecutionException {
966         List<Dependency> flattenedDependencies = new ArrayList<Dependency>();
967         // resolve all direct and inherited dependencies...
968         try {
969             createFlattenedDependencies( effectiveModel, flattenedDependencies );
970         }
971         catch (Exception e) {
972             throw new MojoExecutionException("unable to create flattened dependencies", e);
973         }
974         if ( isEmbedBuildProfileDependencies() )
975         {
976             Model projectModel = this.project.getModel();
977             Dependencies modelDependencies = new Dependencies();
978             modelDependencies.addAll( projectModel.getDependencies() );
979             for ( Profile profile : projectModel.getProfiles() )
980             {
981                 // build-time driven activation (by property or file)?
982                 if ( isBuildTimeDriven( profile.getActivation() ) )
983                 {
984                     List<Dependency> profileDependencies = profile.getDependencies();
985                     for ( Dependency profileDependency : profileDependencies )
986                     {
987                         if ( modelDependencies.contains( profileDependency ) )
988                         {
989                             // our assumption here is that the profileDependency has been added to model because of
990                             // this build-time driven profile. Therefore we need to add it to the flattened POM.
991                             // Non build-time driven profiles will remain in the flattened POM with their dependencies
992                             // and
993                             // allow dynamic dependencies due to OS or JDK.
994                             flattenedDependencies.add( modelDependencies.resolve(profileDependency) );
995                         }
996                     }
997                 }
998             }
999             getLog().debug( "Resolved " + flattenedDependencies.size() + " dependency/-ies for flattened POM." );
1000         }
1001         return flattenedDependencies;
1002     }

1022     /**
1023      * Collects the resolved direct and transitive {@link Dependency dependencies} from the given <code>effectiveModel</code>.
1024      * The collected dependencies are stored in order, so that the leaf dependencies are prioritized in front of direct dependencies.
1025      * In addition, every non-leaf dependencies will exclude its own direct dependency, since all transitive dependencies
1026      * will be collected.
1027 	 *
1028      * Transitive dependencies are all going to be collected and become a direct dependency. Maven should already resolve
1029      * versions properly because now the transitive dependencies are closer to the artifact. However, when this artifact is
1030      * being consumed, Maven Enforcer Convergence rule will fail because there may be multiple versions for the same transitive dependency.
1031 	 *
1032      * Typically, exclusion can be done by using the wildcard. However, a known Maven issue prevents convergence enforcer from
1033      * working properly w/ wildcard exclusions. Thus, this will exclude each dependencies explicitly rather than using the wildcard.
1034      *
1035      * @param projectDependencies is the effective POM {@link Model}'s current dependencies
1036      * @param flattenedDependencies is the {@link List} where to add the collected {@link Dependency dependencies}.
1037      * @throws DependencyTreeBuilderException
1038      * @throws ArtifactDescriptorException
1039      */
1040     private void createFlattenedDependenciesAll( List<Dependency> projectDependencies, List<Dependency> flattenedDependencies )
1041             throws DependencyTreeBuilderException, ArtifactDescriptorException {
1042         final Queue<DependencyNode> dependencyNodeLinkedList = new LinkedList<DependencyNode>() {};
1043         final Set<String> processedDependencies = new HashSet<>();
1044 
1045         final Artifact projectArtifact = this.project.getArtifact();
1046 
1047         final DependencyNode dependencyNode = this.dependencyTreeBuilder.buildDependencyTree(this.project,
1048                 this.localRepository, null);
1049 
1050         dependencyNode.accept(new DependencyNodeVisitor() {
1051             @Override public boolean visit(DependencyNode node) {
1052                 if (node.getArtifact().getGroupId().equals(projectArtifact.getGroupId()) && node.getArtifact().getArtifactId().equals(projectArtifact.getArtifactId())) {
1053                     return true;
1054                 }
1055                 if (node.getState() != DependencyNode.INCLUDED) return true;
1056                 dependencyNodeLinkedList.add(node);
1057                 return true;
1058             }
1059 
1060             @Override public boolean endVisit(DependencyNode node) {
1061                 return true;
1062             }
1063         });
1064 
1065         while (!dependencyNodeLinkedList.isEmpty()) {
1066             DependencyNode node = dependencyNodeLinkedList.poll();
1067 
1068             Artifact artifact = node.getArtifact();
1069 
1070             Dependency dependency = new Dependency();
1071             dependency.setGroupId(artifact.getGroupId());
1072             dependency.setArtifactId(artifact.getArtifactId());
1073             dependency.setVersion(artifact.getVersion());
1074             dependency.setClassifier(artifact.getClassifier());
1075             dependency.setOptional(artifact.isOptional());
1076             dependency.setScope(artifact.getScope());
1077             dependency.setType(artifact.getType());
1078 
1079             List<Exclusion> exclusions = new LinkedList<>();
1080 
1081             org.eclipse.aether.artifact.Artifact aetherArtifact = new DefaultArtifact(artifact.getGroupId(), artifact.getArtifactId(), null, artifact.getVersion());
1082             ArtifactDescriptorRequest request = new ArtifactDescriptorRequest(aetherArtifact, null, null);
1083             ArtifactDescriptorResult artifactDescriptorResult = this.artifactDescriptorReader
1084                     .readArtifactDescriptor(this.session.getRepositorySession(), request);
1085 
1086             for (org.eclipse.aether.graph.Dependency artifactDependency: artifactDescriptorResult.getDependencies()) {
1087                 if ("test".equals(artifactDependency.getScope())) {
1088                     continue;
1089                 }
1090                 Exclusion exclusion = new Exclusion();
1091                 exclusion.setGroupId(artifactDependency.getArtifact().getGroupId());
1092                 exclusion.setArtifactId(artifactDependency.getArtifact().getArtifactId());
1093                 exclusions.add(exclusion);
1094             }
1095 
1096             dependency.setExclusions(exclusions);
1097 
1098             // convert dependency to string for the set, since Dependency doesn't implement equals, etc.
1099             String dependencyString = dependency.getManagementKey();
1100 
1101             if (!processedDependencies.add(dependencyString)) {
1102                 continue;
1103             }
1104 
1105             Dependency flattenedDependency = createFlattenedDependency( dependency );
1106             if (flattenedDependency != null) {
1107                 flattenedDependencies.add(flattenedDependency);
1108             }
1109         }
1110     }
