48  /**
49   * Works around thread safety issues when modifying the global singleton {@link org.apache.maven.model.building.DefaultModelBuilder DefaultModelBuilder}
50   * with custom {@link ProfileInjector} and {@link ProfileSelector}. Instead of modifying the global {@code DefaultModelBuilder}, this class
51   * creates a new {@code DefaultModelBuilder} and equips it with the currently active components like {@link ModelProcessor}, {@link ModelValidator} etc.
52   * which might have been modified/provided by other Maven extensions. 
53   * 
54   * @author Falko Modler
55   * @since 1.2.3
56   */
57  @Named
58  public class ModelBuilderThreadSafetyWorkaround
59  {
60  
61      @Inject
62      private ModelProcessor modelProcessor;
63  
64      @Inject
65      private ModelValidator modelValidator;
66  
67      @Inject
68      private ModelNormalizer modelNormalizer;
69  
70      @Inject
71      private ModelInterpolator modelInterpolator;
72  
73      @Inject
74      private ModelPathTranslator modelPathTranslator;
75  
76      @Inject
77      private ModelUrlNormalizer modelUrlNormalizer;
78  
79      @Inject
80      private SuperPomProvider superPomProvider;
81  
82      @Inject
83      private InheritanceAssembler inheritanceAssembler;
84  
85      @Inject
86      private PluginManagementInjector pluginManagementInjector;
87  
88      @Inject
89      private DependencyManagementInjector dependencyManagementInjector;
90  
91      @Inject
92      private DependencyManagementImporter dependencyManagementImporter;
93  
94      @Inject
95      @Nullable
96      private LifecycleBindingsInjector lifecycleBindingsInjector;
97  
98      @Inject
99      private PluginConfigurationExpander pluginConfigurationExpander;
100 
101     @Inject
102     private ReportConfigurationExpander reportConfigurationExpander;
103 
104     @Inject
105     private ReportingConverter reportingConverter;
106 
107     public ModelBuildingResult build( ModelBuildingRequest buildingRequest, ProfileInjector customInjector, ProfileSelector customSelector )
108         throws ModelBuildingException
109     {
110         // note: there is neither DefaultModelBuilder.get*(), nor DefaultModelBuilder.clone()
111         return new DefaultModelBuilderFactory().newInstance()
112             .setProfileInjector( customInjector )
113             .setProfileSelector( customSelector )
114             // apply currently active ModelProcessor etc. to support extensions like jgitver
115             .setDependencyManagementImporter( dependencyManagementImporter )
116             .setDependencyManagementInjector( dependencyManagementInjector )
117             .setInheritanceAssembler( inheritanceAssembler )
118             .setLifecycleBindingsInjector( lifecycleBindingsInjector )
119             .setModelInterpolator( modelInterpolator )
120             .setModelNormalizer( modelNormalizer )
121             .setModelPathTranslator( modelPathTranslator )
122             .setModelProcessor( modelProcessor )
123             .setModelUrlNormalizer( modelUrlNormalizer )
124             .setModelValidator( modelValidator )
125             .setPluginConfigurationExpander( pluginConfigurationExpander )
126             .setPluginManagementInjector( pluginManagementInjector )
127             .setReportConfigurationExpander( reportConfigurationExpander )
128             .setReportingConverter( reportingConverter )
129             .setSuperPomProvider( superPomProvider )
130             .build( buildingRequest );
131     }
132 }
